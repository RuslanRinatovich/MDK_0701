# 🧪 Практическая работа №25  
## Восстановление БД на новом экземпляре SQL Server

⏱️ **Продолжительность:** 90 минут  
🎯 **Цель работы:**  
Научиться восстанавливать базы данных на другой сервер (или другой экземпляр) с учётом перемещения файлов, работы с разными версиями SQL Server, восстановления зашифрованных бэкапов и исправления одиноких пользователей.

🖥️ **Оборудование:**  
Виртуальная машина с Windows Server / Windows 10, установленными SQL Server Developer Edition и SSMS.

## 🧠 Глубокое погружение: перед началом

В этой работе вы будете имитировать восстановление базы данных на новый экземпляр SQL Server. Поскольку у нас одна виртуальная машина, мы будем использовать:
- Разные имена баз для имитации нового сервера
- Разные папки для имитации другой структуры дисков
- Удаление и создание сертификатов для имитации работы с шифрованием

В результате вы научитесь выполнять все этапы реальной миграции базы данных.

## 📋 Ход выполнения работы

### 🔹 Этап 1. Подготовка исходной базы (15 мин)

1. Запустите **SSMS** и подключитесь к экземпляру SQL Server.

2. Создайте исходную базу данных:

```sql
CREATE DATABASE MigrationSource_Фамилия;
GO
USE MigrationSource_Фамилия;
GO
```

3. Создайте таблицы и наполните их данными:

```sql
CREATE TABLE dbo.Users (
    UserID INT IDENTITY(1,1) PRIMARY KEY,
    LoginName NVARCHAR(50),
    FullName NVARCHAR(100),
    Email NVARCHAR(100),
    CreatedDate DATETIME DEFAULT GETDATE()
);

CREATE TABLE dbo.Orders (
    OrderID INT IDENTITY(1,1) PRIMARY KEY,
    UserID INT REFERENCES dbo.Users(UserID),
    OrderDate DATETIME DEFAULT GETDATE(),
    Amount DECIMAL(10,2),
    Status NVARCHAR(20)
);

INSERT INTO dbo.Users (LoginName, FullName, Email)
VALUES 
    ('ivanov', 'Иван Иванов', 'ivanov@company.com'),
    ('petrov', 'Петр Петров', 'petrov@company.com'),
    ('sidorova', 'Анна Сидорова', 'sidorova@company.com');

INSERT INTO dbo.Orders (UserID, Amount, Status)
VALUES 
    (1, 15000, 'Выполнен'),
    (2, 23000, 'В обработке'),
    (1, 8000, 'Новый'),
    (3, 32000, 'Выполнен');
GO
```

4. Создайте пользователя базы данных (для демонстрации одиноких пользователей):

```sql
CREATE USER TestUser WITHOUT LOGIN;
GRANT SELECT ON dbo.Users TO TestUser;
GO
```

5. Создайте папку `C:\Backup\ПР25\`:

```sql
EXEC xp_cmdshell 'mkdir C:\Backup\ПР25\';  -- если xp_cmdshell включён
```

6. **Создайте полную резервную копию**:

```sql
BACKUP DATABASE [MigrationSource_Фамилия]
TO DISK = N'C:\Backup\ПР25\MigrationSource_Фамилия_Full.bak'
WITH INIT, COMPRESSION, CHECKSUM, STATS = 10;
```

7. **Сделайте скриншот** выполнения бэкапа.

### 🔹 Этап 3. Восстановление на "новый сервер" с MOVE (20 мин)

В этом этапе мы имитируем восстановление на новый сервер с другой структурой дисков.

3.1. **Создайте новую структуру папок** (имитация другого сервера):

```sql
EXEC xp_cmdshell 'mkdir C:\NewServer\Data';
EXEC xp_cmdshell 'mkdir C:\NewServer\Logs';
```

3.2. **Восстановите базу под новым именем с MOVE:**

```sql
RESTORE DATABASE MigrationTarget_Фамилия
FROM DISK = N'C:\Backup\ПР25\MigrationSource_Фамилия_Full.bak'
WITH 
    MOVE 'MigrationSource_Фамилия' TO 'C:\NewServer\Data\MigrationTarget_Фамилия.mdf',
    MOVE 'MigrationSource_Фамилия_log' TO 'C:\NewServer\Logs\MigrationTarget_Фамилия_log.ldf',
    STATS = 10,
    RECOVERY;
```

3.3. **Проверьте восстановленные данные:**

```sql
USE MigrationTarget_Фамилия;
GO
SELECT * FROM dbo.Users;
SELECT * FROM dbo.Orders;
```

3.4. **Проверьте, что файлы создались в новых папках:**

```sql
EXEC xp_cmdshell 'dir C:\NewServer\Data\';
EXEC xp_cmdshell 'dir C:\NewServer\Logs\';
```

3.5. **Сделайте скриншоты** всех шагов.

### 🔹 Этап 4. Проблемы с одинокими пользователями (15 мин)

После восстановления базы на другой сервер пользователи могут быть не связаны с логинами.

4.1. **Найдите одиноких пользователей:**

```sql
USE MigrationTarget_Фамилия;
GO
EXEC sp_change_users_login @Action = 'Report';
```

Вы должны увидеть пользователя `TestUser` без связанного логина.

4.2. **Создайте логин на "новом сервере":**

```sql
USE master;
GO
CREATE LOGIN TestUser WITH PASSWORD = 'TestUser123!';
```

4.3. **Свяжите пользователя с логином:**

```sql
USE MigrationTarget_Фамилия;
GO
EXEC sp_change_users_login 
    @Action = 'Update_One',
    @UserNamePattern = 'TestUser',
    @LoginName = 'TestUser';
```

4.4. **Проверьте, что проблема решена:**

```sql
EXEC sp_change_users_login @Action = 'Report';
-- Должен вернуть пустой результат
```

4.5. **Сделайте скриншоты** отчёта о одиноких пользователях и исправления.

### 🔹 Этап 5. Восстановление зашифрованного бэкапа на другой сервер (15 мин)

В этом этапе мы создадим зашифрованный бэкап и восстановим его на "новый сервер".

5.1. **Создайте сертификат для шифрования (на исходном сервере):**

```sql
USE master;
GO

-- Создаём мастер-ключ (если нет)
IF NOT EXISTS (SELECT * FROM sys.symmetric_keys WHERE name = '##MS_DatabaseMasterKey##')
BEGIN
    CREATE MASTER KEY ENCRYPTION BY PASSWORD = 'MasterKey_Фамилия_2024!';
END
GO

-- Создаём сертификат
CREATE CERTIFICATE MigrationCert
WITH SUBJECT = 'Certificate for migration - ' + @@SERVERNAME;
GO
```

5.2. **Создайте зашифрованную резервную копию:**

```sql
BACKUP DATABASE [MigrationSource_Фамилия]
TO DISK = N'C:\Backup\ПР25\MigrationSource_Фамилия_Encrypted.bak'
WITH 
    COMPRESSION,
    ENCRYPTION (
        ALGORITHM = AES_256,
        SERVER CERTIFICATE = MigrationCert
    ),
    STATS = 10;
```

5.3. **Создайте резервную копию сертификата (для переноса):**

```sql
BACKUP CERTIFICATE MigrationCert
TO FILE = N'C:\Backup\ПР25\MigrationCert.cer'
WITH PRIVATE KEY (
    FILE = N'C:\Backup\ПР25\MigrationCert_PrivateKey.pvk',
    ENCRYPTION BY PASSWORD = N'CertKey_Фамилия_2024!'
);
```

5.4. **Попробуйте восстановить зашифрованный бэкап (должна быть ошибка):**

```sql
RESTORE DATABASE MigrationTarget_Encrypted_Фамилия
FROM DISK = N'C:\Backup\ПР25\MigrationSource_Фамилия_Encrypted.bak'
WITH 
    MOVE 'MigrationSource_Фамилия' TO 'C:\NewServer\Data\MigrationTarget_Encrypted_Фамилия.mdf',
    MOVE 'MigrationSource_Фамилия_log' TO 'C:\NewServer\Logs\MigrationTarget_Encrypted_Фамилия_log.ldf',
    RECOVERY;
```

**Запишите текст ошибки.** Ошибка будет связана с отсутствием сертификата.

5.5. **Восстановите сертификат на "новом сервере":**

```sql
-- Создаём мастер-ключ на "новом сервере" (если нет)
IF NOT EXISTS (SELECT * FROM sys.symmetric_keys WHERE name = '##MS_DatabaseMasterKey##')
BEGIN
    CREATE MASTER KEY ENCRYPTION BY PASSWORD = 'MasterKey_Фамилия_2024!';
END
GO

-- Восстанавливаем сертификат
CREATE CERTIFICATE MigrationCert
FROM FILE = N'C:\Backup\ПР25\MigrationCert.cer'
WITH PRIVATE KEY (
    FILE = N'C:\Backup\ПР25\MigrationCert_PrivateKey.pvk',
    DECRYPTION BY PASSWORD = N'CertKey_Фамилия_2024!'
);
GO
```

5.6. **Повторите восстановление** — теперь оно должно пройти успешно.

5.7. **Проверьте данные:**

```sql
USE MigrationTarget_Encrypted_Фамилия;
GO
SELECT * FROM dbo.Users;
SELECT * FROM dbo.Orders;
```

5.8. **Сделайте скриншоты** ошибки, восстановления сертификата и успешного восстановления.

### 🔹 Этап 6. Автоматизация восстановления (15 мин)

6.1. **Создайте процедуру для автоматического восстановления:**

```sql
USE master;
GO

CREATE OR ALTER PROCEDURE dbo.usp_RestoreDatabaseToNewServer
    @BackupPath NVARCHAR(500),
    @NewDatabaseName NVARCHAR(128),
    @DataPath NVARCHAR(255) = 'C:\NewServer\Data\',
    @LogPath NVARCHAR(255) = 'C:\NewServer\Logs\',
    @OverwriteExisting BIT = 0
AS
BEGIN
    SET NOCOUNT ON;
    
    DECLARE @FileList TABLE (
        LogicalName NVARCHAR(128),
        PhysicalName NVARCHAR(260),
        Type CHAR(1),
        Size INT,
        MaxSize INT,
        FileId INT
    );
    
    DECLARE @SQL NVARCHAR(MAX);
    DECLARE @LogicalDataName NVARCHAR(128);
    DECLARE @LogicalLogName NVARCHAR(128);
    
    -- Проверка существования бэкапа
    IF NOT EXISTS (SELECT 1 FROM sys.master_files WHERE physical_name = @BackupPath)
    BEGIN
        PRINT 'Проверка существования файла...';
    END
    
    -- Получение списка файлов из бэкапа
    BEGIN TRY
        INSERT INTO @FileList
        EXEC ('RESTORE FILELISTONLY FROM DISK = ''' + @BackupPath + '''');
    END TRY
    BEGIN CATCH
        PRINT 'Ошибка при чтении бэкапа: ' + ERROR_MESSAGE();
        RETURN;
    END CATCH
    
    -- Получение логических имён
    SELECT @LogicalDataName = LogicalName 
    FROM @FileList 
    WHERE Type = 'D';
    
    SELECT @LogicalLogName = LogicalName 
    FROM @FileList 
    WHERE Type = 'L';
    
    IF @LogicalDataName IS NULL OR @LogicalLogName IS NULL
    BEGIN
        PRINT 'Не удалось определить логические имена файлов';
        RETURN;
    END
    
    -- Формирование команды восстановления
    SET @SQL = 'RESTORE DATABASE [' + @NewDatabaseName + '] FROM DISK = ''' + @BackupPath + ''''
        + ' WITH MOVE ''' + @LogicalDataName + ''' TO ''' + @DataPath + @NewDatabaseName + '.mdf'''
        + ', MOVE ''' + @LogicalLogName + ''' TO ''' + @LogPath + @NewDatabaseName + '_log.ldf'''
        + ', STATS = 10';
    
    IF @OverwriteExisting = 1
        SET @SQL = @SQL + ', REPLACE';
    
    SET @SQL = @SQL + ', RECOVERY';
    
    -- Вывод информации
    PRINT '=== Параметры восстановления ===';
    PRINT 'Бэкап: ' + @BackupPath;
    PRINT 'Целевая база: ' + @NewDatabaseName;
    PRINT 'Путь к данным: ' + @DataPath;
    PRINT 'Путь к журналам: ' + @LogPath;
    PRINT 'Логическое имя данных: ' + @LogicalDataName;
    PRINT 'Логическое имя журнала: ' + @LogicalLogName;
    PRINT 'Перезапись: ' + CASE @OverwriteExisting WHEN 1 THEN 'Да' ELSE 'Нет' END;
    PRINT '=================================';
    
    -- Выполнение восстановления
    BEGIN TRY
        EXEC sp_executesql @SQL;
        PRINT '✅ Восстановление успешно завершено';
    END TRY
    BEGIN CATCH
        PRINT '❌ Ошибка при восстановлении: ' + ERROR_MESSAGE();
        THROW;
    END CATCH
END;
GO
```

6.2. **Протестируйте процедуру:**

```sql
-- Сначала удалим базу, чтобы проверить восстановление
DROP DATABASE IF EXISTS MigrationTarget_Test_Фамилия;
GO

-- Запускаем процедуру
```

```sql

```
