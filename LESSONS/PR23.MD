# 🧪 Практическая работа №23  
## Анализ истории резервных копий и заданий

⏱️ **Продолжительность:** 90 минут  
🎯 **Цель работы:**  
Научиться извлекать и анализировать информацию из системной базы msdb о резервных копиях и выполнении заданий SQL Agent. Освоить методы мониторинга успешных и неудачных операций, выявления проблемных мест и построения отчётов для аудита.

## 🧠 Глубокое погружение: перед началом

База данных msdb — это "чёрный ящик" SQL Server. В ней хранится всё:
- когда и какие бэкапы делались
- какие задания запускались и с каким результатом
- кто и когда отправлял email
- какие страницы были повреждены

В этой работе вы научитесь читать этот "чёрный ящик" и делать правильные выводы.

### 🔹 Этап 3. Анализ частоты и свежести бэкапов (15 мин)

**Задание:** Определите, все ли базы на сервере регулярно бэкапятся.

3.1. **Последний бэкап для каждой базы:**

```sql
WITH LastBackup AS (
    SELECT 
        database_name,
        type,
        backup_finish_date,
        ROW_NUMBER() OVER (PARTITION BY database_name, type ORDER BY backup_finish_date DESC) AS rn
    FROM msdb.dbo.backupset
    WHERE database_name NOT IN ('master', 'model', 'msdb', 'tempdb')
)
SELECT 
    database_name,
    MAX(CASE WHEN type = 'D' AND rn = 1 THEN backup_finish_date END) AS LastFull,
    MAX(CASE WHEN type = 'I' AND rn = 1 THEN backup_finish_date END) AS LastDiff,
    MAX(CASE WHEN type = 'L' AND rn = 1 THEN backup_finish_date END) AS LastLog,
    DATEDIFF(day, 
        MAX(CASE WHEN type = 'D' AND rn = 1 THEN backup_finish_date END), 
        GETDATE()) AS DaysSinceFull,
    CASE 
        WHEN MAX(CASE WHEN type = 'D' AND rn = 1 THEN backup_finish_date END) IS NULL THEN '❌ НЕТ ПОЛНОГО БЭКАПА'
        WHEN DATEDIFF(day, 
            MAX(CASE WHEN type = 'D' AND rn = 1 THEN backup_finish_date END), 
            GETDATE()) > 7 THEN '⚠️ Более 7 дней'
        ELSE '✅ OK'
    END AS FullBackupStatus
FROM LastBackup
GROUP BY database_name
ORDER BY DaysSinceFull DESC;
```

3.2. **Базы, у которых нет бэкапов вообще:**

```sql
SELECT 
    d.name AS DatabaseName,
    d.create_date,
    d.recovery_model_desc
FROM sys.databases d
LEFT JOIN msdb.dbo.backupset bs ON d.name = bs.database_name
WHERE d.database_id > 4
    AND bs.database_name IS NULL
ORDER BY d.name;
```

3.3. **Размер бэкапов по дням (последние 30 дней):**

```sql
SELECT 
    database_name,
    CONVERT(DATE, backup_start_date) AS BackupDate,
    COUNT(*) AS BackupCount,
    SUM(backup_size / 1048576) AS TotalSizeMB,
    SUM(compressed_backup_size / 1048576) AS TotalCompressedSizeMB,
    AVG(backup_size / 1048576) AS AvgSizeMB
FROM msdb.dbo.backupset
WHERE backup_start_date > DATEADD(day, -30, GETDATE())
    AND database_name = 'MSDBPractice_Фамилия'
GROUP BY database_name, CONVERT(DATE, backup_start_date)
ORDER BY BackupDate DESC;
```

```sql
SELECT 
    j.name AS JobName,
    j.enabled,
    j.description,
    j.date_created,
    j.date_modified,
    MAX(CASE WHEN jh.run_status = 1 THEN '✅' ELSE '❌' END) AS LastStatus,
    MAX(CONVERT(DATETIME, 
        LEFT(jh.run_date, 4) + '-' + 
        SUBSTRING(CAST(jh.run_date AS VARCHAR), 5, 2) + '-' + 
        RIGHT(jh.run_date, 2) + ' ' + 
        LEFT(RIGHT(1000000 + jh.run_time, 6), 2) + ':' + 
        SUBSTRING(RIGHT(1000000 + jh.run_time, 6), 3, 2) + ':' + 
        RIGHT(RIGHT(1000000 + jh.run_time, 6), 2))) AS LastRunTime
FROM msdb.dbo.sysjobs j
LEFT JOIN msdb.dbo.sysjobhistory jh ON j.job_id = jh.job_id
WHERE jh.step_id = 0  -- только итоговые записи по заданию
GROUP BY j.name, j.enabled, j.description, j.date_created, j.date_modified
ORDER BY LastRunTime DESC;
```

### 🔹 Этап 5. Комплексный отчёт по состоянию сервера (20 мин)

**Задание:** Создайте единый отчёт, объединяющий информацию из разных таблиц.

5.1. **Создайте сводную таблицу состояния сервера:**

```sql
-- Временная таблица для результатов
CREATE TABLE #ServerHealth (
    MetricType NVARCHAR(50),
    MetricName NVARCHAR(200),
    MetricValue NVARCHAR(500),
    Status NVARCHAR(20)
);

-- 1. Статистика по бэкапам
INSERT INTO #ServerHealth (MetricType, MetricName, MetricValue, Status)
SELECT 
    'Backup' AS MetricType,
    database_name + ' - Последний полный' AS MetricName,
    ISNULL(CONVERT(VARCHAR, MAX(backup_finish_date), 120), 'НЕТ БЭКАПА') AS MetricValue,
    CASE 
        WHEN MAX(backup_finish_date) IS NULL THEN 'CRITICAL'
        WHEN DATEDIFF(day, MAX(backup_finish_date), GETDATE()) > 7 THEN 'WARNING'
        ELSE 'OK'
    END AS Status
FROM msdb.dbo.backupset
WHERE type = 'D'
    AND database_name NOT IN ('master', 'model', 'msdb', 'tempdb')
GROUP BY database_name;

-- 2. Проваленные задания за последние сутки
INSERT INTO #ServerHealth (MetricType, MetricName, MetricValue, Status)
SELECT 
    'Job' AS MetricType,
    'Failed jobs last 24h' AS MetricName,
    CAST(COUNT(*) AS NVARCHAR) AS MetricValue,
    CASE WHEN COUNT(*) > 0 THEN 'WARNING' ELSE 'OK' END AS Status
FROM msdb.dbo.sysjobhistory
WHERE run_status = 0
    AND run_date >= CONVERT(VARCHAR, DATEADD(day, -1, GETDATE()), 112);

-- 3. Задания, которые не запускались более 7 дней
INSERT INTO #ServerHealth (MetricType, MetricName, MetricValue, Status)
SELECT 
    'Job' AS MetricType,
    'Jobs not run for >7 days' AS MetricName,
    CAST(COUNT(*) AS NVARCHAR) AS MetricValue,
    CASE WHEN COUNT(*) > 0 THEN 'WARNING' ELSE 'OK' END AS Status
FROM (
    SELECT 
        j.name,
        MAX(jh.run_date) AS LastRun
    FROM msdb.dbo.sysjobs j
    LEFT JOIN msdb.dbo.sysjobhistory jh ON j.job_id = jh.job_id
    WHERE j.enabled = 1
    GROUP BY j.name
    HAVING MAX(jh.run_date) IS NULL 
        OR CONVERT(DATETIME, MAX(jh.run_date)) < DATEADD(day, -7, GETDATE())
) t;

-- 4. Размер msdb
INSERT INTO #ServerHealth (MetricType, MetricName, MetricValue, Status)
SELECT 
    'Database' AS MetricType,
    'msdb size (MB)' AS MetricName,
    CAST(SUM(size * 8 / 1024) AS NVARCHAR) AS MetricValue,
    'INFO' AS Status
FROM msdb.sys.database_files;

-- Результат
SELECT 
    MetricType,
    MetricName,
    MetricValue,
    CASE Status
        WHEN 'OK' THEN '✅'
        WHEN 'WARNING' THEN '⚠️'
        WHEN 'CRITICAL' THEN '❌'
        ELSE 'ℹ️'
    END + ' ' + Status AS Status
FROM #ServerHealth
ORDER BY 
    CASE Status
        WHEN 'CRITICAL' THEN 1
        WHEN 'WARNING' THEN 2
        WHEN 'OK' THEN 3
        ELSE 4
    END,
    MetricType,
    MetricName;

DROP TABLE #ServerHealth;
```

### 🔹 Этап 6. Диагностика проблем и поиск узких мест (10 мин)

**Задание:** Найдите проблемы в автоматизации.

6.1. **Базы без полного бэкапа за последние 7 дней:**

```sql
SELECT 
    d.name AS DatabaseName,
    d.recovery_model_desc,
    MAX(bs.backup_finish_date) AS LastFullBackup,
    DATEDIFF(day, MAX(bs.backup_finish_date), GETDATE()) AS DaysSinceLastFull
FROM sys.databases d
LEFT JOIN msdb.dbo.backupset bs ON d.name = bs.database_name AND bs.type = 'D'
WHERE d.database_id > 4
    AND d.state_desc = 'ONLINE'
GROUP BY d.name, d.recovery_model_desc
HAVING MAX(bs.backup_finish_date) IS NULL 
    OR DATEDIFF(day, MAX(bs.backup_finish_date), GETDATE()) > 7
ORDER BY DaysSinceLastFull DESC;
```

6.2. **Задания, которые часто проваливаются:**

```sql
SELECT 
    j.name AS JobName,
    COUNT(*) AS FailCount,
    MAX(jh.run_date) AS LastFailDate,
    MIN(jh.run_date) AS FirstFailDate
FROM msdb.dbo.sysjobs j
JOIN msdb.dbo.sysjobhistory jh ON j.job_id = jh.job_id
WHERE jh.run_status = 0
GROUP BY j.name
HAVING COUNT(*) > 3
ORDER BY FailCount DESC;
```

6.3. **Самая старая запись в истории:**

```sql
SELECT 
    'Старейший бэкап' AS Item,
    MIN(backup_start_date) AS OldestDate
FROM msdb.dbo.backupset
UNION ALL
SELECT 
    'Старейшая запись задания',
    MIN(CONVERT(DATETIME, run_date))
FROM msdb.dbo.sysjobhistory;
```

6.4. **Размер истории (количество записей по годам):**

```sql
SELECT 
    YEAR(backup_start_date) AS Year,
    COUNT(*) AS BackupCount
FROM msdb.dbo.backupset
GROUP BY YEAR(backup_start_date)
ORDER BY Year DESC;
```
